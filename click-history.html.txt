  <script type="module">
    document.getElementById('year').textContent = new Date().getFullYear();
    
    // Import Firebase functions
    import { getAuth, onAuthStateChanged } from 'https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js';

    let currentUser = null;

    // Check if user is logged in
    window.addEventListener('load', () => {
      const auth = getAuth();
      onAuthStateChanged(auth, (user) => {
        if (user) {
          currentUser = user;
        } else {
          // Redirect to login if not authenticated
          window.location.href = 'login.html';
        }
      });
    });

    // Event listeners for buttons
    document.getElementById('refresh-history').addEventListener('click', () => {
      showMessage('History refreshed!', 'success');
    });

    document.getElementById('clear-history').addEventListener('click', () => {
      if (confirm('Are you sure you want to clear all history? This action cannot be undone.')) {
        showMessage('History cleared!', 'success');
      }
    });

    document.getElementById('export-csv').addEventListener('click', () => {
      showMessage('CSV export feature coming soon!', 'info');
    });

    document.getElementById('export-json').addEventListener('click', () => {
      showMessage('JSON export feature coming soon!', 'info');
    });

    document.getElementById('logout-btn').addEventListener('click', async function(e) {
      e.preventDefault();
      try {
        if (typeof smartAuth !== 'undefined') {
          await smartAuth.signOut();
          window.location.href = 'index.html';
        }
      } catch (err) {
        console.error('Logout error:', err);
      }
    });

    function showMessage(text, type = 'info') {
      // Create a temporary message element
      const message = document.createElement('div');
      message.style.cssText = `
        position: fixed;
        top: 20px;
        right: 20px;
        padding: 1rem 1.5rem;
        border-radius: 8px;
        color: white;
        font-weight: 600;
        z-index: 10000;
        transition: all 0.3s ease;
        background: ${type === 'success' ? '#008e45' : type === 'info' ? '#0070f3' : '#6c757d'};
      `;
      message.textContent = text;
      document.body.appendChild(message);

      setTimeout(() => {
        message.style.opacity = '0';
        message.style.transform = 'translateX(100px)';
        setTimeout(() => document.body.removeChild(message), 300);
      }, 3000);
    }
  </script>
